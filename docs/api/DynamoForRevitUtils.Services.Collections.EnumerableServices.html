    Examples
    var numbers = new List<int> { 1, 2, 3, 4, 5 };

Action<IEnumerable<int>> print = x => Console.WriteLine(string.Join(", ", x));
numbers.DoIfHasAny(isEven, print); // Outputs: "2, 4"

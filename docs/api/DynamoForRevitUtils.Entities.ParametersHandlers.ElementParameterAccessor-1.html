    [IsVisibleInDynamoLibrary(false)]
public class ElementParameterAccessor<TParamMap> where TParamMap : IElementParameterMap
    Type Parameters
    Name Description
    TParamMap
    Remarks

    ElementParameterMap<T> must have properties of Type ElementParameter with same names as properties on the ElementWrapper<TParamMap>

    Examples

    How to use on getters and setters of ElementWrapper<TParamMap> properties.

    public string DetailNumber {
    get => ParameterAccessor.GetStringValue();
    set => ParameterAccessor.SetValue(value);
}
